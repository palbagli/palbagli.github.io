	\section{Introducción}

		\begin{frame}
			\frametitle{La Curva de Oferta}
			\begin{mydef}
				\textbf{Cantidad ofrecida ($\mathbf{q^\peq{s}}$):} Cantidad de un bien que los vendedores quieren y pueden vender.
			\end{mydef}
			\begin{mydef}
				\textbf{Función de oferta:} Función que relaciona la cantidad ofrecida de un bien con su precio, manteniendo constantes la tecnología de producción y los precios de los insumos. $$q^\peq{s}=f\left(p,\overline{w},\overline{r},\overline{\text{tecnología}},\overline{\text{otros factores...}}\right)$$
			\end{mydef}
		\end{frame}
		
		\begin{frame}
			\frametitle{La Curva de Oferta}
			\begin{mydef}
				\textbf{Ley de la oferta:} Con todo lo demás constante, la cantidad ofrecida de un bien ($q^\peq{s}$) aumenta cuando su precio ($p$) aumenta.
			\end{mydef}
		\end{frame}
		
		\begin{frame}
			\frametitle{La Curva de Oferta}
			Tradicionalmente la graficamos al revés: $$p^\peq{s}=f^\peq{-1}\left(q,\overline{w},\overline{r},\overline{\text{tecnología}},\overline{\text{otros factores...}}\right)$$
			\begin{figure}[htbp!]
				\centering
				\begin{tikzpicture}[scale=1]
					\draw[<->, thick] (0,4) -- (0,0) -- (4,0);
					\draw plot [domain=0:ln(20)](\x,{exp(\x)/5});
					\node [right] at (3,4) {$S$};
					\node [below] at (4,0) {$q$};
					\node [left] at (0,4) {$p$};
				\end{tikzpicture}
				%\caption{Curva de Oferta}
			\end{figure}
		\end{frame}
		
		\begin{frame}
			\frametitle{La Curva de Oferta}
			De esta manera, la curva de oferta refleja el mínimo precio que un productor estaría dispuesto a aceptar a cambio de vender una unidad adicional. Es decir, el \textbf{costo marginal de producción}. $$\text{Ley de la oferta}\implies\text{la curva de oferta tiene pendiente positiva}$$
		\end{frame}
		
		\begin{frame}
			\frametitle{La Curva de Oferta}
			Es \textbf{muy} importante distinguir entre:
			\begin{mydef}
				\begin{itemize}
					\item \textbf{Cambio en la cantidad ofrecida:} Cuando cambia $p$ y se mantienen constantes otros determinantes de la oferta, se produce un movimiento en/sobre/através de la curva de oferta.
					\item \textbf{Cambio en la oferta:} Cuando cambia alguno de los determinantes de la oferta que no están representados en los ejes del gráfico (es decir, cualquiera distinto de $p$), cambia la cantidad ofrecida a cada precio, por lo que se desplaza la curva de oferta.
				\end{itemize}
			\end{mydef}
		\end{frame}
		
		
	\section{Fundamento Teórico}

		\begin{frame}
			\frametitle{Teoría de la Firma}
			Estudiaremos la decisión de producción de una firma tomadora de precios (tanto en el mercado de su producto como en los mercados de insumos).
			
			Modelaremos a la empresa como una organización que contrata factores productivos ($L$ y $K$, a los que paga $w$ y $r$ respectivamente) que usa como insumos en la producción de un bien y cuyo objetivo es maximizar sus ganancias ($\pi$).
		\end{frame}	

		\begin{frame}
			\frametitle{Teoría de la Firma}
			Definamos algunos conceptos:
			\begin{mydef}
				\begin{itemize}
					\item \textbf{Ingreso total ($\mathbf{IT}$):} Cantidad que la firma recibe por la venta de $q$ unidades a precio $p$. $$IT\left(q\right)=p\cdot q$$
					\item \textbf{Costo total ($\mathbf{CT}$):} Valor de mercado de los insumos que la empresa utiliza en la producción de $q$ unidades. Incluye costos explícitos y costos implícitos.
				\end{itemize}
			\end{mydef}
		\end{frame}	

		\begin{frame}
			\frametitle{Teoría de la Firma}
				\begin{itemize}
					\item \textbf{Costos explícitos:} Costos de los insumos que requieren que la empresa desembolse dinero.
					\item \textbf{Costos implícitos:} Costos de los insumos que \textbf{no} requieren que la empresa desembolse dinero.
					\item \textbf{Utilidad contable:} Ingreso total menos costos explícitos.
					\item \textbf{Utilidad económica:} Ingreso total menos costo total $$\pi\left(q\right)=IT\left(q\right)-CT\left(q\right)$$
				\end{itemize}
		\end{frame}	

		\begin{frame}
			\frametitle{Teoría de la Firma}
				\begin{itemize}
					\item \textbf{Función de producción:} Relación entre la cantidad de insumos utilizados y la cantidad producida (PT, producto total) del bien. $$q=F\left(K,L\right)$$
					\item \textbf{Producto medio ($\mathbf{PMe}$):} Cantidad producida por unidad del factor en promedio. $$PMe Z=\frac{q}{Z}$$
					\item \textbf{Producto marginal ($\mathbf{PMg}$):} Cambio en la producción que se obtiene al usar una unidad adicional del insumo.$$PMg Z=\frac{\Delta q}{\Delta Z}=\frac{F\left(Y,Z+1\right)-F\left(Y,Z\right)}{\left(Z+1\right)-Z}$$			
				\end{itemize}
		\end{frame}	

		\begin{frame}
			\frametitle{Teoría de la Firma}
				Suponemos que PT es creciente en el uso de factores, pero a tasa decreciente. Es decir,
				\begin{itemize}
					\item $F\left(K,L\right)$ es creciente en sus argumentos
					\item $PMg Z$ es decreciente en $Z$
				\end{itemize}
		\end{frame}

		\begin{frame}
			\frametitle{Teoría de la Firma}
			Notar que:
			\begin{itemize}
				\item Cuando $PMgL>0$, $PT$ crece con $L$
				\item La tasa de crecimiento de $PT$ está dada por $PMgL$
				\item $PMgL$ decreciente $\implies$ $PT$ crece a tasa decreciente
				\item Cuando $PMgL>PMeL$, $PMeL$ crece. Cuando $PMgL<PMeL$, $PMeL$ cae y cuando $PMgL=PMeL$, $PMeL$ no cambia.
			\end{itemize}
		\end{frame}	

		\begin{frame}
			\frametitle{Teoría de la Firma}
			¿Qué pasa con el costo total a medida que aumenta la producción?
			\begin{itemize}
				\item $F\rp{K,L} \text{ creciente }\implies$ si aumenta $q$ es porque se está usando más $K$, más $L$, o más de ambos.
				\item La empresa es tomadora de precios en mercados de factores $\implies w \text{ y } r$ están fijos.
				\item Luego, $\Delta^\peq{+}q\implies\Delta^\peq{+}CT\rp{q}$.
			\end{itemize}
		\end{frame}	

		\begin{frame}
			\frametitle{Teoría de la Firma}
			Todavía podemos decir algo más:
			\begin{itemize}
				\item Rendimiento decreciente en los factores ($PMg$ decreciente) $\implies$ la cantidad adicional de insumo necesaria para producir una unidad adicional es cada vez mayor.
				\item En otras palabras, la unidad adicional es cada vez más cara.
				\item Luego, $CT\rp{q}$ crece a tasa creciente a medida que aumenta la producción.
			\end{itemize}
		\end{frame}

			\frametitle{Teoría de la Firma}
			Más conceptos importantes...
			\begin{mydef}
				\begin{itemize}
					\item \textbf{Costo total medio ($\mathbf{CMeT}$):} $CMeT\rp{q}=\frac{CT\rp{q}}{q}$.
					\item \textbf{Costo variable medio ($\mathbf{CMeV}$):} $CMeV\rp{q}=\frac{CV\rp{q}}{q}$.
					\item \textbf{Costo fijo medio ($\mathbf{CMeF}$):} $CMeF\rp{q}=\frac{CF}{q}$.
					\item \textbf{Costo marginal ($\mathbf{CMg}$):} $CMg\rp{q}=\frac{\Delta CT\rp{q}}{\Delta q}=\frac{CT\rp{q+1}-CT\rp{q}}{\rp{q+1}-q}$.
				\end{itemize}
			\end{mydef}
			Desde luego, $$CMeT=CMeF+CMeV$$
		\end{frame}

			\frametitle{Teoría de la Firma}
			\begin{mydef}
				\begin{itemize}
					\item \textbf{Economías de escala:} $CMeT^\peq{LP}$ disminuye cuando aumenta $q$
					\item \textbf{Deseconomías de escala:} $CMeT^\peq{LP}$ aumenta cuando aumenta $q$
					\item \textbf{Economías constantes de escala:} $CMeT^\peq{LP}$ no cambia cuando aumenta $q$
				\end{itemize}
			\end{mydef}
			Recordando que $CMeT\rp{q}=\frac{CT\rp{q}}{q}$ y que $CT\rp{q}$ es una función creciente, notamos que la pregunta relevante es ¿$CT$ aumenta proporcionalmente más, menos o lo mismo que $q$? 
		\end{frame}	
		
		\begin{frame}
			\frametitle{Teoría de la Firma}
			Haremos una última definición antes de modelar la decisión de producción de la firma para obtener su función de oferta
			
			\begin{mydef}
				\textbf{Ingreso marginal ($\mathbf{IMg}$):} Cambio en el ingreso total al producir y vender una unidad adicional.
				$$IMg\rp{q}=\frac{\Delta IT\rp{q}}{\Delta q}=\frac{IT\rp{q+1}-IT\rp{q}}{\rp{q+1}-q}$$
			\end{mydef}
		\end{frame}			
		
		\begin{frame}
			\frametitle{Teoría de la Firma}
			Suponemos que el objetivo de la firma es escoger la cantidad $q=q^\peq{s}$ que maximiza $\pi\rp{q}=IT\rp{q}-CT\rp{q}$. El cambio en la utilidad económica de la empresa al aumentar la producción en una unidad esta dado por $$\Delta \pi = IMg\rp{q}-CMg\rp{q}$$
		\end{frame}					
		
		\begin{frame}
			\frametitle{Teoría de la Firma}
			¿Qué podemos decir de $q^\peq{s}$, la cantidad que maximiza las ganancias? Notar que:
			\begin{itemize}
				\item $CT\rp{q} \text{ creciente}\implies CMg\rp{q}\geq0$.
				\item Si $IMg\rp{q_\peq{0}}>CMg\rp{q_\peq{0}}$ (lo que significa que $IMg\rp{q_\peq{0}}>0$), el aumento del ingreso total causado por un aumento de la producción es mayor que el aumento en el costo total. En $q_\peq{0}$ es posible incrementar las ganancias aumentando la producción.
			\end{itemize}
		\end{frame}
			
		\begin{frame}
			\frametitle{Teoría de la Firma}
			\begin{itemize}
				\item Si $IMg\rp{q_\peq{1}}<CMg\rp{q_\peq{1}}$ y 
				\begin{enumerate}[(a)]
					\item $IMg\rp{q_\peq{1}}<0$, al disminuir la producción aumenta el ingreso total y se reduce el costo total.
					\item $IMg\rp{q_\peq{1}}>0$, al disminuir la producción, la reducción del ingreso total es menor que la reducción del costo total.
				\end{enumerate}
				En ambos casos es posible aumentar las ganancias reduciendo la producción a una menor que $q_\peq{1}$.
			\end{itemize}
		\end{frame}	
			
		\begin{frame}
			\frametitle{Teoría de la Firma}
			Luego,
			\begin{itemize}
				\item $q^\peq{s}$ no puede ser una cantidad como $q_\peq{0}$. Es decir, $$IMg\rp{q^\peq{s}}\ngtr CMg\rp{q^\peq{s}}$$
				\item $q^\peq{s}$ no puede ser una cantidad como $q_\peq{1}$. Es decir, $$IMg\rp{q^\peq{s}}\nless CMg\rp{q^\peq{s}}$$
			\end{itemize}
			Por lo tanto, $q^\peq{s}$ debe ser una cantidad tal que $$IMg\rp{q^\peq{s}}=CMg\rp{q^\peq{s}}$$
		\end{frame}				
		
		\begin{frame}
			\frametitle{Teoría de la Firma}
			\textbf{Conclusión:} La firma maximiza ganancias escogiendo la cantidad cuyo costo marginal de producción es igual a su ingreso marginal.
			Notar que esta conclusión es más general que el caso de la firma competitiva (tomadora de precios).
		\end{frame}

			\textbf{Conclusión:} La empresa tomadora de precios maximiza ganancias escogiendo la cantidad cuyo costo marginal de producción es igual al precio de mercado. $$CMg\rp{q^\peq{s}}=p$$
			En otras palabras, la oferta de la empresa competitiva es su curva de costo marginal. \textbf{Pero...}

		\begin{frame}
			\frametitle{Teoría de la Firma}
			¿Qué pasa si la cantidad que iguala $CMg\rp{q^\peq{s}}=p$ genera un ingreso total que no es suficiente para cubrir los costos evitables?
			\begin{itemize}
				\item \textbf{CP:} Aunque cese temporalmente la producción, los costos fijos están comprometidos y deben pagarse. Son inevitables. La firma cierra si no cubre sus costos medios variables.
				\item \textbf{LP:} Existe la posibilidad de salir del mercado y cesar todos los compromisos. Todos los costos son evitables. La firma abandona el mercado si no cubre sus costos medios totales.
			\end{itemize}
		\end{frame}

	\section{La Curva de Oferta}

		\begin{frame}
			\frametitle{Determinantes de la Oferta}
				\begin{itemize}
					\item \textbf{Precios de los insumos:} $\Delta^\peq{+}p_\peq{z}\implies\Delta^\peq{+}CMg\rp{q}$
					\item \textbf{Tecnología:} Avances tecnológicos permiten producir lo mismo usando menos recursos $\implies\Delta^\peq{-}CMg\rp{q}$
					\item \textbf{Expectativas:} Si se espera $\Delta^\peq{+}p_\peq{x}$ en el futuro, hay incentivo a almacenar parte de la producción para posponer la venta. Del mismo modo, las expectativas respecto de $p_\peq{z}$ y/o tecnología alteran incentivos.
					\item \textbf{Cantidad de vendedores:} Si hay más vendedores, aumenta la oferta de mercado
				\end{itemize}
		\end{frame}

		\begin{frame}
			\frametitle{Elasticidad de la Oferta}
			\begin{mydef}
				\textbf{Elasticidad precio de la oferta:} Una medida de cuánto responde $q^\peq{s}$ respecto al cambio en $p$. $$\varepsilon_\peq{q,p}\equiv\frac{\Delta\%q^\peq{s}}{\Delta\%p}$$
			\end{mydef}
		\end{frame}

		\begin{frame}
			\frametitle{Elasticidad de la Oferta}
			Notar que
			\begin{align*}
				\frac{\Delta\%q^\peq{s}}{\Delta\%p}&=\frac{\Delta q^\peq{s}/q^\peq{s}}{\Delta p/p} \\
																					 &=\frac{1}{\Delta p / \Delta q^\peq{s}}\cdot\frac{p}{q^\peq{s}}								
			\end{align*}
		\end{frame}

		\begin{frame}
			\frametitle{Elasticidad de la Oferta}
			$\text{Ley de la oferta }\implies\varepsilon_\peq{q,p}\geq0$
			\begin{itemize}
				\item $\varepsilon_\peq{q,p}>1\implies$ oferta elástica ($\Delta\%q^\peq{s}>\Delta\%p$).
				\item $\varepsilon_\peq{q,p}=1\implies$ oferta de elasticidad unitaria ($\Delta\%q^\peq{s}=\Delta\%p$).
				\item $\varepsilon_\peq{q,p}<1\implies$ oferta inelástica ($\Delta\%q^\peq{s}<\Delta\%p$).
			\end{itemize}
		\end{frame}

		\begin{frame}
			\frametitle{Elasticidad de la Oferta}
			Ejemplo: oferta lineal $$p=a+b\cdot q^\peq{s}$$ con $b>0$. 
			
			\vspace{.3cm}
			¿Qué podemos decir sobre $\varepsilon_\peq{q,p}$?
			\begin{itemize}
				\item $a>0\implies\varepsilon_\peq{q,p}>1$
				\item $a=0\implies\varepsilon_\peq{q,p}=1$
				\item $a<0\implies\varepsilon_\peq{q,p}<1$
			\end{itemize}
		\end{frame}

		\begin{frame}
			\frametitle{Elasticidad de la Oferta}
			Si calculamos el cambio porcentual como $$\Delta\%x=\frac{x_\peq{final}-x_\peq{inicial}}{x_\peq{inicial}}$$ obtendremos resultados distintos dependiendo del punto de partida. Por eso usamos el método del punto medio $$\Delta\%x=\frac{x_\peq{final}-x_\peq{inicial}}{\rp{\frac{x_\peq{inicial}+x_\peq{final}}{2}}}$$
		\end{frame}

		\begin{frame}
			\frametitle{Elasticidad de la Oferta}
			Al haber libre entrada y salida,
				\begin{itemize}
					\item Si $\pi\rp{q^\peq{s}\rp{p}}>0\implies$ entran firmas, desplazando la oferta de corto plazo. $p$ cae.
					\item Si $\pi\rp{q^\peq{s}\rp{p}}<0\implies$ salen firmas, desplazando la oferta de corto plazo. $p$ sube.
				\end{itemize}
			Por lo tanto, en equilibrio $$\pi=0$$
		\end{frame}

			\frametitle{Elasticidad de la Oferta}
			Pero $$\pi=\rp{p-CMeT}\cdot q$$
			Luego, $$\pi=0 \text{ y } q>0\implies p=CMeT$$
			También sabemos que $p=CMg$. Por lo tanto, debe ser cierto que $$CMeT=CMg$$
			Lo que ocurre en la escala eficiente. Concluimos que $$p=CMeT_\peq{min}$$
		\end{frame}

	\section{Análisis de Bienestar}

		\begin{frame}
			\frametitle{Excedente del Productor}
			\begin{mydef}
				\textbf{Excedente del productor ($\mathbf{EP}$):} Diferencia entre lo que recibe un vendedor por un bien y el costo en que incurre para proporcionarlo.
			\end{mydef}
		\end{frame}	
	
		\begin{frame}
			\frametitle{Excedente del Productor}
			Ejemplo:
			\begin{itemize}
				\item Sea $c_\peq{i}$ el costo en que incurre el individuo $i$ para vender una unidad del bien $x$.
				\item Cada individuo vende a lo más una unidad $\implies$ la decisión es vender o no.
				\item Sea $p$ el precio al que se puede vender una unidad.
			\end{itemize}
		\end{frame}	
			
		\begin{frame}
			\frametitle{Excedente del Productor}
			Decisión racional: $$\text{Si}\left\{
					\begin{array}{ll}
						p<c_\peq{i} & \text{no vende} \\
						p=c_\peq{i} & \text{está indiferente} \\
						p>c_\peq{i}	& \text{vende}
					\end{array}
				\right.$$
			\textit{Nota:}  supondremos que, en caso de indiferencia, vende. Este supuesto es irrelevante para nuestras conclusiones.
		\end{frame}	
		
		\begin{frame}
			\frametitle{Excedente del Productor}
			Supongamos que hay 4 vendedores:
			\begin{table}[htbp!]
				\centering
				\resizebox{4cm}{!}{
					\begin{tabular}{|c|c|}\hline
						$i$&$c_\peq{i}$ \\  \hline 
						 A & \$500/u 		\\ 
						 B & \$600/u 		\\ 
						 C & \$800/u 		\\ 
						 D & \$900/u 		\\ \hline
					\end{tabular}}
			\end{table}
		\end{frame}	

		\begin{frame}
			\frametitle{Excedente del Productor}
			Caso 1: Se licita la compra de 1 unidad del bien $x$ (se comprará al vendedor que ofrezca el menor precio).
			\begin{itemize}
				\item Precio cae rápidamente hasta que $A$ ofrece $\$\rp{600-\delta}$ con $\delta>0$ y pequeño. (¿Por qué no hasta $c_\peq{A}=\$500$?)
				\item $A$ vende su unidad de $x$ a $\$600$ y le costó $\$500\implies$ su excedente es $EP_\peq{A}^\peq{caso 1}=\$100/u\cdot1u=\$100$.
			\end{itemize}
			\textit{Nota:} pensemos en $\delta$ muy pequeño, es decir $\delta\rightarrow0$.
		\end{frame}	

		\begin{frame}
			\frametitle{Excedente del Productor}
			Caso 2: Se licita la compra de 2 unidades, ambas al mismo precio (pero a distintos vendedores).
			\begin{itemize}
				\item Subasta termina cuando $A$ y/o $B$ ofrece $\$\rp{800-\delta}$ con $\delta>0$ y pequeño. 
				\item $EP_\peq{A}^\peq{caso 2}=\rp{\$800/u-\$500/u}\cdot1u=\$300$.
				\item $EP_\peq{B}^\peq{caso 2}=\rp{\$800/u-\$600/u}\cdot1u=\$200$.
				\item $EP_\peq{Total}^\peq{caso 2}=\$200+\$300=\$500$.
			\end{itemize}
			\textit{Nota:} Estamos pensando de nuevo en $\delta\rightarrow0$.
		\end{frame}

			\frametitle{Excedente del Productor}

			\begin{itemize}
				\item El costo marginal es el cambio en el costo variable:$$
					\begin{array}{ll}
						CMg\rp{q}	&= CT\rp{q+1} - CT\rp{q} \\
											&= CV\rp{q+1}+CF-\rp{CV\rp{q}+CF} \\
											&= CV\rp{q+1} - CV\rp{q} \\
					\end{array}$$
				\item Luego, el costo variable es el área bajo la curva de costo marginal.
			\end{itemize}

			\frametitle{Excedente del Productor}
			Sin embargo,
			\begin{itemize}
				\item $\Delta EP = \Delta\pi$:$$
					\begin{array}{ll}
					\Delta \pi &= \pi\rp{q^\peq{s}_\peq{1}} - \pi\rp{q^\peq{s}_\peq{0}} \\
										 &= IT\rp{q^\peq{s}_\peq{1}}-CV\rp{q^\peq{s}_\peq{1}}-CF-\rp{IT\rp{q^\peq{s}_\peq{0}}-CV\rp{q^\peq{s}_\peq{0}}-CF}\\
										 &= IT\rp{q^\peq{s}_\peq{1}}-CV\rp{q^\peq{s}_\peq{1}} - \rp{IT\rp{q^\peq{s}_\peq{0}}-CV\rp{q^\peq{s}_\peq{0}}} \\
										 &= EP_\peq{1}-EP_\peq{0}\\
										 &= \Delta EP
					\end{array}$$ 
			\end{itemize}
